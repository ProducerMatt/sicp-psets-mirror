(define (find-discrete-log y g p)
  (define (internal-x-iter x)
    (cond ((> x (- p 2)) false)
          ((= (modulo y p) (expmod g x p)) x)
          (else
	    (internal-x-iter (+ x 1)))))
  (internal-x-iter 2))
